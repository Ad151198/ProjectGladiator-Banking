Create table CUSTOMER_DETAILS (
CUSTOMER_ID NUMBER(20) NOT NULL PRIMARY KEY,
CUSTOMER_NAME VARCHAR2(255) NOT NULL,
DOB VARCHAR2(255) NOT NULL,
GENDER VARCHAR2(10) NOT NULL,
CONTACT_NUMBER NUMBER(10) NOT NULL,
OCCUPATION VARCHAR2(255) NOT NULL,
ANNUAL_INCOME NUMBER(20) NOT NULL,
EMAIL_ID VARCHAR2(255) NOT NULL,
RESIDENTIAL_ADDRESS_ID NUMBER(20) NOT NULL REFERENCES ADDRESS_DETAILS(ADDRESS_ID),
PERMANENT_ADDRESS_ID NUMBER(20) NOT NULL REFERENCES ADDRESS_DETAILS(ADDRESS_ID)
)



CREATE TABLE ADDRESS_DETAILS(
ADDRESS_ID number(20) NOT NULL PRIMARY KEY,
ADDRESS_LINE_1 VARCHAR2(255) NOT NULL,
ADDRESS_LINE_2 VARCHAR2(255),
LANDMARK VARCHAR2(30),
CITY VARCHAR2(30) NOT NULL,
STATE VARCHAR2(30) NOT NULL,
COUNTRY VARCHAR2(30) NOT NULL,
ZIPCODE number(6) NOT NULL
)




CREATE TABLE ACCOUNT_DETAILS(
ACCOUNT_NUMBER NUMBER(20) NOT NULL PRIMARY KEY,
CUSTOMER_ID NUMBER(20) NOT NULL REFERENCES CUSTOMER_DETAILS(CUSTOMER_ID),
ACCOUNT_TYPE VARCHAR2(30) NOT NULL,
AVAILABLE_BALANCE NUMBER(20) NOT NULL,
IFSC VARCHAR2(30) NOT NULL REFERENCES BRANCH_DETAILS(IFSC)
)


CREATE TABLE BRANCH_DETAILS(
IFSC VARCHAR2(30) NOT NULL PRIMARY KEY,
ADDRESS_ID NUMBER(20) NOT NULL REFERENCES ADDRESS_DETAILS(ADDRESS_ID),
BRANCH_NAME VARCHAR2(30) NOT NULL
)

CREATE TABLE LOGIN_CREDENTIALS(
CUSTOMER_ID NUMBER(20) NOT NULL PRIMARY KEY REFERENCES CUSTOMER_DETAILS(CUSTOMER_ID),
LOGIN_PASSWORD VARCHAR2(30) NOT NULL,
TRANSACTION_PIN NUMBER(20) NOT NULL
)

CREATE TABLE IDENTITY_DOCUMENT(
CUSTOMER_ID NUMBER(20) NOT NULL PRIMARY KEY REFERENCES CUSTOMER_DETAILS(CUSTOMER_ID),
PAN VARCHAR2(10) NOT NULL,
AADHAAR_NUMBER NUMBER(12) NOT NULL
)
