Create table CUSTOMER_DETAILS (
CUSTOMER_ID NUMBER(20)  PRIMARY KEY,
CUSTOMER_NAME VARCHAR2(255) ,
DOB VARCHAR2(255) ,
GENDER VARCHAR2(10) ,
CONTACT_NUMBER NUMBER(10) ,
OCCUPATION VARCHAR2(255) ,
ANNUAL_INCOME NUMBER(20) ,
EMAIL_ID VARCHAR2(255) ,
RESIDENTIAL_ADDRESS_ID NUMBER(20)  REFERENCES ADDRESS_DETAILS(ADDRESS_ID),
PERMANENT_ADDRESS_ID NUMBER(20)  REFERENCES ADDRESS_DETAILS(ADDRESS_ID)
)



CREATE TABLE ADDRESS_DETAILS(
ADDRESS_ID number(20)  PRIMARY KEY,
ADDRESS_LINE_1 VARCHAR2(255) ,
ADDRESS_LINE_2 VARCHAR2(255),
LANDMARK VARCHAR2(30),
CITY VARCHAR2(30) ,
STATE VARCHAR2(30) ,
COUNTRY VARCHAR2(30) ,
ZIPCODE number(6) 
)




CREATE TABLE ACCOUNT_DETAILS(
ACCOUNT_NUMBER NUMBER(20)  PRIMARY KEY,
CUSTOMER_ID NUMBER(20)  REFERENCES CUSTOMER_DETAILS(CUSTOMER_ID),
ACCOUNT_TYPE VARCHAR2(30) ,
AVAILABLE_BALANCE NUMBER(20) ,
IFSC VARCHAR2(30)  REFERENCES BRANCH_DETAILS(IFSC)
)


CREATE TABLE BRANCH_DETAILS(
IFSC VARCHAR2(30)  PRIMARY KEY,
ADDRESS_ID NUMBER(20)  REFERENCES ADDRESS_DETAILS(ADDRESS_ID),
BRANCH_NAME VARCHAR2(30) 
)

CREATE TABLE LOGIN_CREDENTIALS(
CUSTOMER_ID NUMBER(20)  PRIMARY KEY REFERENCES CUSTOMER_DETAILS(CUSTOMER_ID),
LOGIN_PASSWORD VARCHAR2(30) ,
TRANSACTION_PIN NUMBER(20) 
)

CREATE TABLE IDENTITY_DOCUMENT(
CUSTOMER_ID NUMBER(20)  PRIMARY KEY REFERENCES CUSTOMER_DETAILS(CUSTOMER_ID),
PAN VARCHAR2(10) ,
AADHAAR_NUMBER NUMBER(12) 
)
